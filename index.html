    @docker run --rm -v `pwd`:/ws -u `id -u`:`id -g` \
        $(CODENARC) python3 /opt/run_codenarc.py -- \
            -includes='**/Jenkinsfile,**/*.groovy,**/*.dsl' \
            -rulesetfiles=file\:ruleset.groovy
